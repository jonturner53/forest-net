00050 inline void vnetTbl::addVnet(vnet_t vn) {
00051 // Add the specified vnet to the set of vnets in use.
00052 // Use bit 0 of links field to denote a valid vnet.
00053         if (vn > 0 && vn <= maxv) tbl[vn].links = 1;
00054         tbl[vn].plnk = 0;
00055 }

00057 inline void vnetTbl::removeVnet(vnet_t vn) {
00058 // Remove the specified vnet from the set of vnets in use.
