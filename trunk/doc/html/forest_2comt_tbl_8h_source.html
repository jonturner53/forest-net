00081 inline void comtTbl::addLink(int ctte, int lnk, bool rflg, bool cflg) {
00082 // Add the link to the set of valid links for ctte.
00083 // Rflg is true if far end of link is another router.
00084 // Cflg is true if far end of link is a core router for this comtree.
00085         if (ctte > 0 && ctte <= maxte && valid(ctte)) {
00086                 tbl[ctte].links |= (1 << lnk);
00087                 if (rflg) tbl[ctte].rlinks |= (1 << lnk);
00088                 if (rflg && cflg) tbl[ctte].clinks |= (1 << lnk);
00089         }
00090 }

00092 inline void comtTbl::removeLink(int ctte, int lnk) {
00093 // Remove the link from the set of valid links for ctte.
00094         if (ctte > 0 && ctte <= maxte && valid(ctte)) {
00095                 tbl[ctte].links &= ~(1 << lnk);
00096                 tbl[ctte].rlinks &= ~(1 << lnk);
00097                 tbl[ctte].clinks &= ~(1 << lnk);
00098         }

00148 inline int comtTbl::links(int ctte, uint16_t* lnks, int limit) const {
00149 // Return all the comtree links for the specified comtree.
00150 // Note that this requires a comtree table entry number.
00151 // Lnks points to a vector that will hold the links on return.
00152 // The limit argument specifies the maximum number of links that
00153 // may be returned. If the table entry specifies more links, the
00154 // extra ones will be ignored. The value returned by the function
00155 // is the number of links that were returned.
00156         if (ctte < 1 || ctte > maxte || !valid(ctte)) return 0;
00157         return listLinks(tbl[ctte].links,lnks,limit);
00158 }
