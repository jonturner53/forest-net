00104 inline int rteTbl::link(int te) const {
00105 // Return link for a unicast table entry.
00106 // Return 0 if this is a multicast entry.
00107         assert(valid(te));
00108         if (forest::mcastAdr(tbl[te].adr)) return 0;
00109         return tbl[te].lnks;

00121 inline bool rteTbl::addLink(int te, int lnk) {
00122 // Add a link to a multicast table entry. Return true on success,
00123 // false if entry is not a multicast entry.
00124         assert(valid(te));
00125         if (forest::ucastAdr(tbl[te].adr)) return false;
00126         tbl[te].lnks |= (1 << lnk);

00130 inline bool rteTbl::removeLink(int te, int lnk) {
00131 // Remove a link from a multicast table entry. Return true on success,
00132 // false if entry is not a multicast entry.

00140 // Return the hashkey for the specified comtree and address
