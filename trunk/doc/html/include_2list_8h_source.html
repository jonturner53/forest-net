00003 // Invariant properties

00005 // 0 <= first <= N, 0 <= last <= N, -1 <= next[i] <= N
00006 // if first != Null then next[first] != -1
00007 // if last != Null then next[last] = Null
00008 // if next[i] > 0 then next[next[i]] != -1
