00109 // with specified source IP (address,port) pair and the given
00110 // source LFS address. If no match, return Null.
00111 inline int lnkTbl::lookup(int intf, ipa_t pipa, ipp_t pipp, ipa_t slfs) {
00112         ipa_t x = (pipp != LFS_PORT ? slfs : pipa);
00113         int te = ht->lookup(hashkey(pipa,x));
00114         if (te !=0 && intf == interface(te) && 
00115             (pipp == peerPort(te) || peerPort(te) == 0))
00116                 return te;
00117         return 0;
